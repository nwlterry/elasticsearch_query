#!/usr/bin/env bash
set -euo pipefail

# =============================================================================
# Elasticsearch ILM Report + Policies Export (JSON + Focused CSV)
# - Index list with sizes & current ILM status
# - Optional export of all ILM policies: full JSON + focused CSV
# =============================================================================

echo "Script started at $(date)"

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo -e "\n"

# === Configuration ===
ES_PROTO="http"                 # change to "https" if needed
ES_HOST="${ES_PROTO}://localhost:9200"
# Add -k to curl commands below if using self-signed HTTPS certificates

echo "=== Connecting to: ${ES_HOST} ==="

# ─────────────────────────────────────────────────────────────────────────────
echo -e "\n[1/2] Fetching index sizes..."

sizes_http=$(curl -s -u "${ES_USER}:${ES_PASS}" \
  -o sizes.json -w "%{http_code}" \
  "${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open")

echo "Sizes fetch HTTP status: ${sizes_http}"

if [ "${sizes_http}" -ne 200 ] || [ ! -s sizes.json ]; then
  echo "ERROR: Failed to fetch index sizes (HTTP ${sizes_http})"
  head -n 10 sizes.json 2>/dev/null || echo "(empty)"
  exit 1
fi

echo "sizes.json size: $(wc -c < sizes.json) bytes"

# ─────────────────────────────────────────────────────────────────────────────
echo -e "\n[2/2] Building index + ILM report..."

echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE" > report.tsv

jq -r '.[] | [.index, .["store.size"] // 0, .["pri.store.size"] // 0] | @tsv' sizes.json |
while IFS=$'\t' read -r idx total pri; do
  ilm_resp=$(curl -s -u "${ES_USER}:${ES_PASS}" "${ES_HOST}/${idx}/_ilm/explain?human" 2>/dev/null)

  if [[ -z "${ilm_resp}" ]]; then
    policy="unmanaged"
    phase="(fetch failed)"
  else
    policy=$(echo "${ilm_resp}" | jq -r '.indices."'"${idx}"'".policy // "unmanaged"')
    phase=$(echo "${ilm_resp}" | jq -r '.indices."'"${idx}"'".phase // (if .indices."'"${idx}"'".managed // false then "managed-no-phase" else "not managed" end)')
  fi

  echo -e "${idx}\t${total}\t${pri}\t${policy}\t${phase}" >> report.tsv
done

# Display sorted table
tail -n +2 report.tsv | sort -k2 -n -r | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) | column -t -s $'\t'

echo -e "\nIndex report complete."

# ─────────────────────────────────────────────────────────────────────────────
echo -e "\n=== ILM Policies Export (JSON + Focused CSV) ===\n"

read -p "Export all ILM policies (JSON + focused CSV with usage)? (y/N): " export_choice

if [[ "${export_choice,,}" =~ ^(y|yes)$ ]]; then
  echo "Fetching all ILM policies..."

  timestamp=$(date +"%Y-%m-%d_%H-%M-%S")
  json_file="ilm_policies_export_${timestamp}.json"
  csv_file="ilm_policies_export_${timestamp}_focused.csv"

  # Fetch policies
  curl_http=$(curl -s -u "${ES_USER}:${ES_PASS}" \
    -o "${json_file}" -w "%{http_code}" \
    "${ES_HOST}/_ilm/policy?pretty&human")

  if [ "${curl_http}" -ne 200 ] || [ ! -s "${json_file}" ]; then
    echo "ERROR: Failed to fetch policies (HTTP ${curl_http})"
    head -n 10 "${json_file}" 2>/dev/null || echo "(empty)"
    exit 1
  fi

  policy_count=$(jq 'length' "${json_file}" 2>/dev/null || echo "0")
  echo "Found ${policy_count} policies"

  # ────────────── Focused CSV ──────────────
  echo "Generating focused CSV..."

  jq -r --arg today "$(date +%Y-%m-%d)" '
    to_entries[]
    | .key as $name
    | .value
    | if . == null or .policy == null then
      [$name, "null", "null", "none", "-", "-", "-", "-"]
    else
      [
        $name,
        (.version // "null"),
        (.modified_date // "null"),
        (.policy.phases.hot.actions.rollover | tojson // "none"),
        (.policy.phases.cold.min_age // "-"),
        (.policy.phases.cold.actions | keys | join(", ") // "-"),
        (.policy.phases.delete.min_age // "-"),
        (.policy.phases.delete.actions | keys | join(", ") // "-")
      ]
    end | @csv
  ' "$$   {json_file}" > "   $${csv_file}.tmp"

  # Add header
  {
    echo "policy_name,version,modified_date,hot_rollover_conditions,cold_phase_min_age,cold_phase_actions,delete_phase_min_age,delete_phase_actions,used_by_indices_count,used_by_data_streams_count,used_by_indices_list,used_by_data_streams_list"
    tail -n +2 "${csv_file}.tmp"
  } > temp.csv && mv temp.csv "${csv_file}.tmp"

  # ────────────── Add usage information ──────────────
  echo "Collecting usage info (indices + data streams)..."

  curl -s -u "${ES_USER}:${ES_PASS}" \
    "${ES_HOST}/_cat/indices?format=json&h=index,ilm.policy" > indices.json

  curl -s -u "${ES_USER}:${ES_PASS}" \
    "${ES_HOST}/_data_stream" > data_streams.json

  jq -r --slurpfile policies "${json_file}" --slurpfile ind indices.json --slurpfile ds data_streams.json '
    $policies[0] as $pols
    | $ind[0] as $inds
    | $ds[0].data_streams as $dstreams

    | $pols | to_entries[]
    | .key as $policy
    | {
        policy: $policy,
        ind_count: ($inds | map(select(.["ilm.policy"] == $policy)) | length),
        ind_list: ($inds | map(select(.["ilm.policy"] == $policy) | .index) | join(", ") | if length > 200 then .[0:197]+"..." else . end),
        ds_count: ($dstreams | map(select(.ilm_policy == $policy)) | length),
        ds_list: ($dstreams | map(select(.ilm_policy == $policy) | .name) | join(", ") | if length > 200 then .[0:197]+"..." else . end)
      }
    | [
        .policy,
        ($pols[.policy].version // "null"),
        ($pols[.policy].modified_date // "null"),
        ($pols[.policy].policy.phases.hot.actions.rollover | tojson // "none"),
        ($pols[.policy].policy.phases.cold.min_age // "-"),
        ($pols[.policy].policy.phases.cold.actions | keys | join(", ") // "-"),
        ($pols[.policy].policy.phases.delete.min_age // "-"),
        ($pols[.policy].policy.phases.delete.actions | keys | join(", ") // "-"),
        .ind_count,
        .ds_count,
        .ind_list,
        .ds_list
      ] | @csv
  ' "${json_file}" > "${csv_file}"

  # Final header + data
  {
    echo "policy_name,version,modified_date,hot_rollover_conditions,cold_phase_min_age,cold_phase_actions,delete_phase_min_age,delete_phase_actions,used_by_indices_count,used_by_data_streams_count,used_by_indices_list,used_by_data_streams_list"
    cat "${csv_file}"
  } > temp.csv && mv temp.csv "${csv_file}"

  rm -f "${csv_file}.tmp" indices.json data_streams.json

  echo ""
  echo "Export finished:"
  echo "  • Full JSON:          ${json_file}"
  echo "  • Focused CSV:        ${csv_file}"
  echo ""
  echo "Open ${csv_file} in Excel / Google Sheets"
else
  echo "Export skipped."
fi

# ─────────────────────────────────────────────────────────────────────────────
echo -e "\n=== Finished ===\n"

read -p "Remove temporary files (y/N)? " cleanup
if [[ "${cleanup,,}" =~ ^(y|yes)$ ]]; then
  rm -f sizes.json report.tsv ilm_policies_export_* indices.json data_streams.json 2>/dev/null
  echo "Temporary files removed."
fi
