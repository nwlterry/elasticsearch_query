#!/usr/bin/env bash
set -euo pipefail

# ================================================
# Elasticsearch ILM + Indices Size Report + ILM Policies Export
# - Shows index names, total/primary bytes, ILM policy & phase
# - Sorted by total size descending
# - Prompts for credentials
# - Optional: Export all ILM policies JSON
# ================================================

echo "Script started at $(date)"

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo -e "\n"

# Change to https:// if your cluster requires TLS
# Add -k to curl if self-signed cert (insecure!)
ES_PROTO="http"
# ES_PROTO="https"
ES_HOST="${ES_PROTO}://localhost:9200"

echo "=== Using endpoint: ${ES_HOST} ==="

# ────────────────────────────────────────────────
echo -e "\n[1/2] Fetching index sizes..."

sizes_http=$(curl -s -u "${ES_USER}:${ES_PASS}" \
  -o sizes.json -w "%{http_code}" \
  "${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open")

echo "Sizes HTTP status: ${sizes_http}"

if [ "${sizes_http}" -ne 200 ] || [ ! -s sizes.json ]; then
  echo "ERROR: Sizes fetch failed (HTTP ${sizes_http})"
  head -n 10 sizes.json || echo "(file empty/missing)"
  exit 1
fi

echo "sizes.json size: $(wc -c < sizes.json) bytes"

# ────────────────────────────────────────────────
echo -e "\n[2/2] Fetching ILM info per index..."

# Output table header
echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE" > report.tsv

jq -r '.[] | [.index, .["store.size"] // .store_size // 0, .["pri.store.size"] // .pri_store_size // 0] | @tsv' sizes.json |
while IFS=$'\t' read -r idx total pri; do
  echo "DEBUG: Processing ${idx} ..."

  ilm_resp=$(curl -s -u "${ES_USER}:${ES_PASS}" "${ES_HOST}/${idx}/_ilm/explain?human" 2>/dev/null)

  if [[ -z "${ilm_resp}" ]]; then
    policy="unmanaged"
    phase="(fetch failed)"
  else
    policy=$(echo "${ilm_resp}" | jq -r '.indices."'"${idx}"'".policy // "unmanaged"')
    phase=$(echo "${ilm_resp}" | jq -r '.indices."'"${idx}"'".phase // (if .indices."'"${idx}"'".managed // false then "managed-no-phase" else "not managed" end)')
  fi

  echo -e "${idx}\t${total}\t${pri}\t${policy}\t${phase}" >> report.tsv
done

# Sort by size desc (skip header) and display
tail -n +2 report.tsv | sort -k2 -n -r | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) | column -t -s $'\t'

echo -e "\nReport complete. ${#} indices processed."

# ────────────────────────────────────────────────
echo -e "\n=== ILM Policies Export Session ==="

read -p "Export all ILM policies with full config (JSON) for lookup/backup? (y/N): " export_choice
if [[ "${export_choice,,}" == "y" || "${export_choice,,}" == "yes" ]]; then
  echo "Fetching all ILM policies..."

  timestamp=$(date +"%Y-%m-%d_%H-%M-%S")
  export_file="ilm_policies_export_${timestamp}.json"

  policies_http=$(curl -s -u "${ES_USER}:${ES_PASS}" \
    -o "${export_file}" -w "%{http_code}" \
    "${ES_HOST}/_ilm/policy?pretty&human")

  if [ "${policies_http}" -eq 200 ] && [ -s "${export_file}" ]; then
    policy_count=$(jq 'length' "${export_file}" 2>/dev/null || echo "unknown")
    echo "Success! Exported ${policy_count} policies to: ${export_file}"
    echo "Policy names:"
    jq 'keys[]' "${export_file}" 2>/dev/null | sort || echo "(could not list names)"
    echo ""
    echo "You can view it with: cat ${export_file} | jq ."
    echo "Or re-apply one policy: curl -X PUT -u user:pass -H 'Content-Type: application/json' '${ES_HOST}/_ilm/policy/my-policy' -d @ilm_policies_export_....json"
  else
    echo "ERROR: Export failed (HTTP ${policies_http})"
    head -n 10 "${export_file}" || echo "(file empty)"
    rm -f "${export_file}" 2>/dev/null
  fi
else
  echo "Export skipped."
fi

echo -e "\n=== Done ===\n"

# Cleanup prompt
read -p "Remove temporary files (sizes.json, report.tsv)? (y/N): " cleanup
if [[ "${cleanup,,}" == "y" || "${cleanup,,}" == "yes" ]]; then
  rm -f sizes.json report.tsv
  echo "Temp files removed."
fi
