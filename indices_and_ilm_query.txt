#!/usr/bin/env bash
set -euo pipefail

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo ""

ES_HOST="localhost:9200"  # try "https://localhost:9200" if needed

echo "Fetching sizes..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  "http://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open" > sizes.json

if [ ! -s sizes.json ]; then
  echo "sizes.json is empty"
  exit 1
fi

echo "Response type:"
jq 'type' sizes.json || echo "jq failed to parse"

echo "Number of indices:"
jq 'length' sizes.json 2>/dev/null || echo "Not an array"

echo "First index (if exists):"
jq '.[0]' sizes.json 2>/dev/null || echo "No elements or invalid"

# Try to print table even if partial
jq -r '
  if type == "array" then
    .[]
    | [.index // "?", .["store.size"] // .store_size // 0, .["pri.store.size"] // .pri_store_size // 0]
    | @tsv
  else
    "Response is not array: " + (tostring)
  end
' sizes.json \
  | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES"; cat -) \
  | column -t -s $'\t'
