#!/usr/bin/env bash
set -euo pipefail

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo ""

ES_HOST="localhost:9200"   # change if needed, e.g. to your-domain:9200 or IP

echo "Fetching index sizes in bytes..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  -o sizes.json \
  "http://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open"

if [ ! -s sizes.json ]; then
  echo "ERROR: sizes.json is empty (0 bytes). Check credentials, host, or network."
  echo "Manual test command:"
  echo "  curl -u \"${ES_USER}:***\" \"http://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b\""
  exit 1
fi

echo "sizes.json fetched ($(wc -c < sizes.json | awk '{print $1}') bytes)"
echo "First few bytes of response (for debug):"
head -c 300 sizes.json | cat -v || true
echo ""

# Detect key style and set variables
if grep -q '"store_size"' sizes.json; then
  TOTAL_KEY="store_size"
  PRI_KEY="pri_store_size"
  echo "Detected snake_case keys"
elif grep -q '"store.size"' sizes.json; then
  TOTAL_KEY="store.size"
  PRI_KEY="pri.store.size"
  echo "Detected dotted keys (store.size / pri.store.size) – using correct jq syntax"
else
  echo "WARNING: No recognizable store size keys found → sizes will be 0"
  TOTAL_KEY="store.size"  # fallback
  PRI_KEY="pri.store.size"
fi

# Extract with proper dotted-key support in jq
jq -r --arg t "$TOTAL_KEY" --arg p "$PRI_KEY" '
  .[] 
  | [.index, (.[$t] // 0), (.[$p] // 0)] 
  | @tsv
' sizes.json \
  | while IFS=$'\t' read -r idx total_bytes pri_bytes; do
      # Skip if index is empty/malformed
      [ -z "$idx" ] && continue

      echo "DEBUG: Processing index $idx (bytes: $total_bytes / $pri_bytes)" >&2

      phase_info=$(curl -s -u "${ES_USER}:${ES_PASS}" \
        "http://${ES_HOST}/$idx/_ilm/explain?pretty" \
        | jq -r --arg idx "$idx" '
            (.indices[$idx] // {managed: false}) |
            if .managed then
              [.policy // "none", .phase // "-"] | join("\t")
            else
              "unmanaged\t-"
            end' 2>/dev/null || echo "unmanaged\t- (error)")

      echo -e "$idx\t$total_bytes\t$pri_bytes\t$phase_info"
    done \
  | sort -k2 -n -r \
  | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) \
  | column -t -s $'\t'

echo ""
echo "Done. If still no rows appear below header:"
echo "- Check DEBUG lines above for processing info"
echo "- Run: jq '.[0]' sizes.json   → paste the output here for further debug"
echo "- Or try without &expand_wildcards=open if indices are closed/hidden"

# rm -f sizes.json  # uncomment to auto-clean
