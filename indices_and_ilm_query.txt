#!/usr/bin/env bash
set -euo pipefail

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo ""

ES_HOST="localhost:9200"   # ← change to https:// if your cluster uses TLS

echo "Fetching index sizes..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  -o sizes.json \
  -w "%{http_code}" \
  "https://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open" > http_code.txt

HTTP_CODE=$(cat http_code.txt | tr -d '\n')

if [ "$HTTP_CODE" != "200" ]; then
  echo "ERROR: HTTP $HTTP_CODE — not OK response"
  echo "Raw first lines of response:"
  head -n 10 sizes.json
  echo "Fix: Check credentials, try https://, or verify with Kibana/console."
  exit 1
fi

if [ ! -s sizes.json ]; then
  echo "ERROR: sizes.json empty"
  exit 1
fi

# Test if it's an array
ARRAY_CHECK=$(jq 'if type == "array" then "ok" else "not-array: " + type end' sizes.json)

if [[ "$ARRAY_CHECK" != *"ok"* ]]; then
  echo "ERROR: Response is not a JSON array → jq will fail"
  echo "Type: $ARRAY_CHECK"
  echo "First 300 chars:"
  head -c 300 sizes.json | cat -v
  echo ""
  echo "Likely an error message instead of data. Check auth/permissions."
  exit 1
fi

echo "Fetching ILM explain..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  -o ilm.json \
  "https://${ES_HOST}/_ilm/explain?human"

echo "Combining data (with null-safety)..."

jq -r --slurpfile ilm ilm.json '
  . as $sizes
  | $ilm[0].indices as $ilm_data
  | ($ilm_data | to_entries | map(select(.value.managed == true)) | length) as $managed_count
  | "DEBUG: Found \($managed_count) managed indices",
    $sizes[]
    | .index as $idx
    | [$idx,
       (.["store.size"] // .store_size // 0),
       (.["pri.store.size"] // .pri_store_size // 0),
       ($ilm_data[$idx].policy // "unmanaged"),
       ($ilm_data[$idx].phase // (if $ilm_data[$idx].managed // false then "unknown-phase" else "not managed" end))
      ] | @tsv
' sizes.json 2> jq_error.log \
  | sort -k2 -n -r \
  | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) \
  | column -t -s $'\t'

if [ -s jq_error.log ]; then
  echo ""
  echo "jq had warnings/errors:"
  cat jq_error.log
fi

echo ""
echo "If still issues, paste:"
echo "- The DEBUG line"
echo "- Output of: jq '.[0]' sizes.json   (first index object)"
echo "- Or run manual GET _cat/indices?format=json&h=store.size"
