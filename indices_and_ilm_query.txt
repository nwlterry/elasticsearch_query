#!/usr/bin/env bash
set -euo pipefail

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo ""

ES_HOST="localhost:9200"

curl -s -u "${ES_USER}:${ES_PASS}" \
  "http://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open" > sizes.json

if [ ! -s sizes.json ] || ! grep -q "store_size" sizes.json; then
  echo "Error: sizes.json is empty or missing store_size keys."
  head -n 20 sizes.json
  exit 1
fi

# Use awk for more reliable tab-separated parsing
cat sizes.json | jq -r '.[] | [.index, .store_size // 0, .pri_store_size // 0] | @tsv' \
  | awk -F'\t' '{
      idx = $1
      total = $2
      pri   = $3

      # Fetch ILM (you can keep this part)
      cmd = "curl -s -u \"${ES_USER}:${ES_PASS}\" \"http://${ES_HOST}/" idx "/_ilm/explain?pretty\""
      cmd | getline ilm_json
      close(cmd)

      phase_info = "unmanaged\t-"
      if (ilm_json != "") {
        # Very basic check – improve with jq if needed
        if (index(ilm_json, "\"managed\": true") > 0) {
          match(ilm_json, /"policy":\s*"([^"]+)"/, arr); policy = arr[1] ? arr[1] : "none"
          match(ilm_json, /"phase":\s*"([^"]+)"/, arr2); phase = arr2[1] ? arr2[1] : "-"
          phase_info = policy "\t" phase
        }
      }

      print idx "\t" total "\t" pri "\t" phase_info
    }' \
  | sort -k2 -n -r \
  | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) \
  | column -t -s $'\t'

# rm sizes.json   # uncomment to clean up
