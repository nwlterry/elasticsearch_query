#!/usr/bin/env bash
set -euo pipefail

read -p "Elasticsearch username: " ES_USER
read -s -p "Elasticsearch password: " ES_PASS
echo ""

ES_HOST="localhost:9200"   # ← change to https:// if needed, add -k to curl if self-signed

echo "Fetching sizes..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  -o sizes.json \
  "https://${ES_HOST}/_cat/indices?format=json&h=index,store.size,pri.store.size&bytes=b&expand_wildcards=open"

echo "Fetching all ILM explain at once..."
curl -s -u "${ES_USER}:${ES_PASS}" \
  -o ilm.json \
  "https://${ES_HOST}/_ilm/explain?human"

if [ ! -s sizes.json ] || [ ! -s ilm.json ]; then
  echo "One of the files is empty — check auth/host/response."
  exit 1
fi

echo "Combining data..."

jq -r --slurpfile ilm ilm.json '
  . as $sizes
  | $ilm[0].indices as $ilm_data
  | $sizes[]
  | .index as $idx
  | [$idx,
     (.store.size // 0),
     (.pri.store.size // 0),
     ($ilm_data[$idx].policy // "unmanaged"),
     ($ilm_data[$idx].phase // (if $ilm_data[$idx].managed then "no-phase" else "unmanaged (no phase)" end))
    ] | @tsv
' sizes.json \
  | sort -k2 -n -r \
  | (echo -e "INDEX\tSTORE_BYTES\tPRI_STORE_BYTES\tILM.POLICY\tPHASE"; cat -) \
  | column -t -s $'\t'

echo ""
echo "Notes:"
echo "- If POLICY is 'unmanaged' for everything → your indices have no ILM attached (common if not using data streams/templates)."
echo "- To attach ILM to existing index: PUT /your-index/_settings { \"index.lifecycle.name\": \"your-policy-name\" }"
echo "- Run GET _ilm/explain?pretty manually to verify."
